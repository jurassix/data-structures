### The Set object lets you store unique values of any type, whether primitive values or object references.
---
```javascript
var set = new Set();

set.add(1); // Set [ 1 ]
set.add(5); // Set [ 1, 5 ]
set.add(5); // Set [ 1, 5 ]
```
---
```javascript
var set = new Set();

// objects are referential equality (===)
set.add({a: 1}); // Set [ {a: 1} ]
set.add({a: 1}); // Set [ {a: 1}, {a: 1} ]
```

// in-place
const dedup = list => {
  for (let i = 0; i < list.length; i++) {
    let item = list[i];
    if (list.indexOf(item) !== i) {
      list.splice(i--, 1);
    }
  }
